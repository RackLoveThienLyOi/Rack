    if not unlocked then return end
    playClick()
    farming = not farming
    if farming then
        FarmBtn.Text = "Auto Farm: ON"
        FarmBtn.BackgroundColor3 = Color3.fromRGB(50,200,50)

local function getGingerbreads()
    local list = {}
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and v.Name:lower():find("ginger") then
            table.insert(list,v)
        end
    end
    return list
end

task.spawn(function()
    while task.wait(0.3) do
        if farming and unlocked then
            Character = LocalPlayer.Character
            local hrp = Character and Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                for _,g in pairs(getGingerbreads()) do
                    if not farming then break end
                    if g and g.Parent then
                        hrp.CFrame = g.CFrame + Vector3.new(0,2,0)
                        task.wait(0.15)
                    end
                end
            end
        end
    end
end)
